    组件 - 弹出层

    使用说明
                                
                            <link rel="stylesheet" type="text/css" href="layer.css" />
                            
                            
                            <script type="text/javascript" src="jquery.js"></script>
                            <script type="text/javascript" src="jquery.layer.js"></script>
    通用接口
    alert confirm iframe 
                                /**
                             * @func alert
                             * @param <String, JQuery Object> 内容
                             * @param <String> 标题
                             * @param <String, Number> 宽度
                             * @param <String, Number> 高度
                             */
                            $.layer.alert('
    弹出层内容
    ', '标题', '280');
                            
                            /**
                             * @func confirm
                             * @param <String, JQuery Object> 内容
                             * @param <Function> 确认按钮回调函数，接收3个参数：$('.ui-layer-main')、弹出层索引、$('.ui-layer-confirm')
                             * @param <String> 标题
                             * @param <String, Number> 宽度
                             * @param <String, Number> 高度
                             */
                            $.layer.confirm('
    10 * 10 =  ？
    ', function(main, index, button){
                                if(main.find(':text').val() == '100'){
                                    return true;
                                }
                            }, '标题', '280');
                            
                            /**
                             * @func iframe
                             * @param <String> url地址
                             * @param <String> 标题
                             * @param <String, Number> 宽度
                             * @param <String, Number> 高度
                             * @desc 跨域无法正常使用
                             * @desc 谷歌浏览器在服务器环境下才能正常使用
                             */
                            $.layer.iframe('iframe/index.html', '标题', 1100);
                            
                            /**
                             * @func tips
                             * @param <String, JQuery Object> 内容
                             * @param <String> 箭头方向 top right bottom left
                             * @param <Object> 显示位置 {top:value, left:value}
                             * @param <Function> 显示回调函数，接收2个参数：$('.ui-layer-main')、弹出层索引
                             */
                            $.layer.tips('
    这是一个弹出层！
    ', type, {}, function(main, index){
                                //code...
                            });

                                isMove:false,

                                    text:'标题',

                                    },

                                        enable:true,

                                    }
                                });
                            });

    API
    普通属性 值类型 默认值 可填值 功能说明
    width Number | String 380 正整数 | 'auto' 宽度
    height Number | String 220 正整数 | 'auto' 高度
    maxHeight Number 0 正整数 最大高度，当高度设置为auto时有效
    content String | Object 空字符串 任意字符串 | jQuery对象 显示内容
    theme String 空字符串 任意字符串 标识、主题
    padding Number 50 整数 弹出层最大尺寸距离window边界距离
    isFadein Boolean false true | false 是否淡入方式显示
    isMove Boolean true true | false 是否可以拖动
    isMask Boolean true true | false 是否有遮罩层
    isClickMask Boolean false true | false 点击遮罩层是否关闭弹出层
    isMoveMask Boolean false true | false 是否有移动遮罩层
    isClose Boolean true true | false 是否能被layerHide()方法关闭
    isCenter Boolean false true | false 是否居中
    isMaxSize Boolean false true | false 是否自适应最大尺寸显示
    isTips Boolean false true | false 是否以tips形式显示
    isSticky Boolean false true | false 点击layer是否置顶
    isFixed Boolean false true | false 是否固定弹出层
    isAnimate Boolean false true | false 是否开启弹出层动画
    对象属性 属性名 值类型 默认值/参数 可填值/参数值 功能说明
    title enable Boolean true true | false 是否启用标题
    text String 空字符串 任意字符串 标题名称
    iframe enable Boolean false true | false 是否载入浮动框架
    src String 空字符串 任意字符串 浮动框架url地址
    offset isBasedPrev Boolean false true | false 是否基于上一个弹出层进行定位
    top Number 0 正整数 上边距
    left Number 0 正整数 左边距
    arrow enable Boolean false true | false 是否启用小箭头
    dir String top top | right | bottom | left 小箭头方向
    close enable Boolean true true | false 是否启用右上角关闭按钮
    text String × 任意字符串 按钮文本内容
    callback Function main $('.ui-layer-main') 点击触发回调函数
    index 弹出层索引
    button $('.ui-layer-close')
    confirm enable Boolean false true | false 是否启用底部确认按钮
    text String 确认 任意字符串 按钮文本内容
    callback Function main $('.ui-layer-main') 点击触发回调函数，当返回值为true时关闭弹出层
    index 弹出层索引
    button $('.ui-layer-confirm')
    cancel enable Boolean false true | false 是否启用底部取消按钮
    text String 取消 任意字符串 按钮文本内容
    callback Function main $('.ui-layer-main') 点击触发回调函数
    index 弹出层索引
    button confirm:{}, cancel:{}, 自定义id:{}, ... Obejct - - 可自定义配置底部按钮，配置后将覆盖cancel和confirm，包含text属性和callback方法，不用配置enable属性
    普通方法 参数 默认值 功能说明
    onShowEvent main $('.ui-layer-main') 弹出层显示时执行
    index 弹出层索引
    onHideEvent main $('.ui-layer-main') 弹出层关闭时执行
    index 弹出层索引
    onResizeEvent main $('.ui-layer-main') 弹出层大小、位置改变后执行
    index 弹出层索引
    onWinRisizeEvent layer $('.ui-layer') window窗口改变大小时执行
    width window窗口宽度
    height window窗口高度
    onWinScrollEvent layer $('.ui-layer') window窗口滚动时执行
    scrollTop window窗口滚动高度
    全局方法 参数 值类型 参数说明 功能说明
    layerResize index | theme Number | String 弹出层索引或者标识 弹出层居中、自适应高度
    iframe undefined | Object 框架DOM对象
    layerHide index | theme Number | String 弹出层索引或者标识 删除弹出层
    layerConfig options JSON Object 配置属性 弹出层全局配置
